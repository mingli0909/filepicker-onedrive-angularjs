1 /**
  2  * OneDrive Controller
  3  * @constructor
  4  * @param {Object} $scope - The title of the book.
  5  * @param {Object Array} $element - The author of the book.
  6  * @param {Object} $attrs - The author of the book.
  7  */
  8 function OneDriveController($scope, $element, $attrs) {
  9     // OneDrive Options
 10     var odOptions = {
 11         clientId: $attrs.attrClientId,
 12         sourceInputElementId: "fileUploadControl",
 13         openInNewWindow: true,
 14         action: '',
 15         success: function(files) { },
 16         progress: function(percent) { /* progress handler */ },
 17         cancel: function() { /* cancel handler */ },
 18         error: function(error) { /* error handler */ }
 19     }   
 20 
 21     /**
 22      * Open Modal
 23      * @constructor
 24      */
 25     $scope.openModal = function() {
 26       var modal = document.getElementById("myModal");
 27       modal.style.display = "block";
 28     };
 29     /**
 30      * Close Modal
 31      * @constructor
 32      */
 33     $scope.closeModal = function() {
 34         var modal = document.getElementById("myModal");
 35         modal.style.display = "none";
 36     }
 37     /**
 38      * Launch OneDrive according to Action
 39      * @constructor
 40      * @param {string} action - Action for 'Save' or 'Download'
 41      */
 42     $scope.launchOneDrive = function(action) { 
 43         odOptions.action = action;
 44 
 45         if(action == 'save')
 46         {
 47             odOptions.success = function (files) { alert('success') };
 48             OneDrive.save(odOptions);
 49         }
 50             
 51         else if(action == 'download')
 52         {
 53             odOptions.success = function (files) { 
 54                 files.value.forEach(function(value) {
 55                     downloadFile(value['@microsoft.graph.downloadUrl'], value['name']);
 56                 })
 57             };
 58             OneDrive.open(odOptions);
 59         }
 60     }
 61     /**
 62      * Download File
 63      * @constructor
 64      * @param {string} uri - Download File Url.
 65      * @param {string} filename - Filename.
 66      */
 67     function downloadFile(uri, filename) {
 68         var a = document.createElement("a");
 69         a.href = uri;
 70         a.setAttribute("download", filename);
 71         a.click();
 72     }
 73 }
