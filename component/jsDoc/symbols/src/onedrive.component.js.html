1 /**
  2  * OneDrive Controller
  3  * @constructor
  4  * @param {Object} $scope - The title of the book.
  5  * @param {Object Array} $element - The author of the book.
  6  * @param {Object} $attrs - The author of the book.
  7  */
  8 function OneDriveController($scope, $location, $element, $attrs, oneDriveService) {
  9 
 10     var params = getUrlParams(window.location.hash);
 11 
 12     if(params['access_token'] != '' && params['access_token'] != null)
 13     {
 14         $scope.authorized = true;
 15         $scope.access_token = params['access_token'];
 16         console.log($scope.access_token);
 17         $scope.current_folder_id = 'root';
 18         getData('root');    
 19     }
 20     /**
 21      * Sign into OneDrive
 22      */
 23     $scope.signIn = function() {
 24         var loginUri = 'https://login.microsoftonline.com/common/oauth2/v2.0/authorize?';
 25         loginUri += `client_id=${$attrs.attrClientId}`;
 26         loginUri += '&scope=files.readwrite offline_access';
 27         loginUri += '&response_type=token';
 28         loginUri += '&redirect_uri=http://localhost:3000';
 29 
 30         window.location.href = loginUri;
 31     }
 32     /**
 33      * Sign out OneDrive
 34      */
 35     $scope.signOut = function() {
 36         localStorage.clear();
 37         $scope.authorized = false;
 38         window.location.href = 'http://localhost:3000';
 39     }
 40     /**
 41      * Triggered after clicking folder with item id
 42      * @param {*} item_id
 43      */
 44     $scope.openFolder = function(item_id) {
 45     
 46         $scope.current_folder_id = item_id;
 47 
 48         if(item_id != 'root')
 49             item_id = `items/${item_id}`;
 50 
 51         oneDriveService.getAllChildren(item_id, $scope.access_token, function(response) {
 52             $scope.files = oneDriveService.parseData(response);
 53         });
 54     }
 55     /**
 56      * Triggered when clicked Go Back Button
 57      */
 58     $scope.goBack = function() {
 59         oneDriveService.getParentId($scope.current_folder_id, $scope.access_token, function(response) {
 60             if(response['parentReference']['path'] == '/drive/root:')
 61                 $scope.openFolder('root');
 62             else
 63                 $scope.openFolder(response['parentReference']['id']);
 64         })
 65     }
 66     /**
 67      * Upload to selected file to OneDrive
 68      */
 69     $scope.uploadToDrive = function() {
 70         var f = document.getElementById('fileUploadControl').files[0];
 71         oneDriveService.uploadFile($scope.current_folder_id, $scope.access_token, f, function(response) {
 72             if(response.status == '201')
 73             {
 74                 alert('Created!');
 75                 $scope.openFolder($scope.current_folder_id);
 76             }
 77         });
 78     }
 79     /**
 80      * Parse queries in the URL
 81      * @param {*} hash 
 82      */
 83     function getUrlParams(hash) {
 84         hash = decodeURIComponent(decodeURIComponent(hash))
 85         var vars = {};
 86         var parts = hash.replace(/[#!#&]+([^=&]+)=([^&]*)/gi, function(m,key,value) {
 87             vars[key] = value;
 88         });
 89         
 90         return vars;
 91     }
 92     /**
 93      * Get OneDrive data with Path
 94      * @param {*} path
 95      */
 96     function getData(path) {
 97         oneDriveService.getAllChildren(path, $scope.access_token, function(response) {
 98             $scope.files = oneDriveService.parseData(response);
 99         });
100     }
101 }
